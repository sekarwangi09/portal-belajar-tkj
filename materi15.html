SQL: Query Dasar

-- Membuat tabel
CREATE TABLE siswa(
  id INT PRIMARY KEY,
  nama VARCHAR(100),
  kelas VARCHAR(20)
);

-- Menambah data
INSERT INTO siswa(id,nama,kelas) VALUES (1,'Ani','XI TKJ 1');

-- Mengambil data
SELECT * FROM siswa;

Gunakan perintah ini di MySQL atau SQLite untuk praktik database. (SQL tidak bisa dijalankan langsung di browser.)
